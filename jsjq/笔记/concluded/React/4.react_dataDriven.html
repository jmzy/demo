import React, { Component } from 'react';
import './App.css';
import Person from './dataDrivenDemo/Person';

class App extends Component {
  state = {
    persons: [
      {name:'apple'}
    ],
    otherState:'anything'
  } 

  myChange = (event) => {
    this.setState({
      persons: [
        {name:event.target.value}
      ],
    })
  }

  render() {
    return (
      <div className="App">
        <Person myChange={this.myChange} name={this.state.persons[0].name}/>
      </div>
    );
  }
}

export default App;
条件判断(普通的if else) 
class App extends Component {
  state = {
    persons: [
      {name:'apple'}
    ],
    personShow:false
  } 

  myChange = (event) => {
    this.setState({
      persons: [
        {name:event.target.value}
      ],
    })
  }

  togglePerson = () => {
    const toggle = this.state.personShow;
    this.setState({
      personShow:!toggle
    })
  }

  render() {
    return (
      <div className="App">
        <h3>hello world</h3>
        <button onClick={this.togglePerson}>切换</button>
        {
          this.state.personShow?
          <Person myChange={this.myChange} name={this.state.persons[0].name}/>
          :null
        }
      </div>
    );
  }
}

JavaScript一种直译式脚本语言，是一种动态类型、弱类型、基于原型的语言，内置支持类型。它的解释器被称为JavaScript引擎，为浏览器的一部分，广泛用于客户端的脚本语言，最早是在HTML（标准通用标记语言下的一个应用）网页上使用，用来给HTML网页增加动态功能。 为了取得技术优势，微软推出了JScript，CEnvi推出ScriptEase，与JavaScript同样可在浏览器上运行。为了统一规格，因为JavaScript兼容于ECMA标准，因此也称为ECMAScript。 在1995年时，由Netscape公司的Brendan Eich，在网景导航者浏览器上首次设计实现而成。因为Netscape与Sun合作，Netscape管理层希望它外观看起来像Java，因此取名为JavaScript。但实际上它的语法风格与Self及Scheme较为接近。

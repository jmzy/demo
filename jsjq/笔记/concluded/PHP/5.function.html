func_get_args() 以数组形式返回所有传参 <br/> func_num_args() 获得传参的总数量 <br/> func_get_arg(number) 获得对应的传参(以0开始数) 
function func(){
	print_r(func_get_args());
};
func('a','b','c','d','f');
//Array ( [0] => a [1] => b [2] => c [3] => d [4] => f )

function funcT(){
	$arr = func_get_args();
	foreach($arr as $key=>$value){
		echo "字母".($key+1)."、字母是{$value}<br/>";
	};
	echo '1共'.func_num_args().'个'.'<br/>';
	echo '第3位字母是'.func_get_arg(2).'<br/>';
};
funcT('a','b','c','d','f');
/*
字母1、字母是a
字母2、字母是b
字母3、字母是c
字母4、字母是d
字母5、字母是f
1共5个
第3位字母是c
 */

加上一个停止调用的条件、否则会陷入死循环的调用 
function jc($i){
	if($i>1){//条件 避免死循环
		$s = $i * jc($i-1);
	}else{
		$s = $i;
	}
	return $s;
};
echo jc(3);

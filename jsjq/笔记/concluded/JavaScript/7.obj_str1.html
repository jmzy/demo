<script>

    //获得每1个字符

indexOf()方法可返回某个指定的字符串值在字符串中首次出现的位置。 
<script>
var str="Hello world!"
console.log(str.indexOf("Hello"));//0  从0开始算、存在
console.log(str.indexOf("World"));//-1 不存在、大小写区分
console.log(str.indexOf("world"));//6  正数代表存在、负数只有-1
<script>

<script>
strObj1.concat(strObj2);

<script>

slice() 抽出字符串对应的字符 
slice() 1个传参值为切割、两个传参值为保留第1个和第2个传参应之间的对象
★这方法也对数组、返回对象string为string、Array为Array

<script>
//string
var a = "123456789";
var b = a.slice(1);
var c = a.slice(1,5);
document.write(b + "<br/>");//23456789
document.write(c + "<br/>");//2345 //★从1开始
document.write(typeof b + "-" + typeof c + "<br/>");//string-string

//Array
var arr = new Array("abc",111,"bcd",222);
var arr1 = arr.slice(1);
var arr2 = arr.slice(2,4);
document.write(arr1 + "<br/>");//111,bcd,222
document.write(arr2 + "<br/>");//bcd,222 //★从1开始、数组不包括自己本身
document.write(typeof arr1 + "-" + typeof arr2 + "<br/>");
//object-object
document.write((arr.constructor === Array)+ "<br/>");//true
document.write((arr1.constructor === Array)+ "<br/>");//true
document.write(arr2.constructor === Array);//true
<script>
substr() 剪切(往后)、指定的子串的开始位置和长度进行切割、0开始
substring() 剪切(往前)、指定的子串的开始位置和-长度进行切割、0开始 
单独1个传参无区别、
两个参数、substr()是往后剪切、substring()是往前剪切

<script>
var str1 = "abcdefghijk";
console.log(str1.substr(2));//"cdefghijk"
console.log(str1.substr(3));//"defghijk"
console.log(str1.substr(3,2));//de"
console.log(str1.substr(3,3));//"def"

var str2 = "abcdefghijk";
console.log(str2.substring(2));//"cdefghijk"
console.log(str2.substring(3));//"defghijk"
console.log(str2.substring(3,2));//"c"
console.log(str2.substring(3,1));//"bc"
<script>

JavaScript一种直译式脚本语言，是一种动态类型、弱类型、基于原型的语言，内置支持类型。它的解释器被称为JavaScript引擎，为浏览器的一部分，广泛用于客户端的脚本语言，最早是在HTML（标准通用标记语言下的一个应用）网页上使用，用来给HTML网页增加动态功能。 为了取得技术优势，微软推出了JScript，CEnvi推出ScriptEase，与JavaScript同样可在浏览器上运行。为了统一规格，因为JavaScript兼容于ECMA标准，因此也称为ECMAScript。 在1995年时，由Netscape公司的Brendan Eich，在网景导航者浏览器上首次设计实现而成。因为Netscape与Sun合作，Netscape管理层希望它外观看起来像Java，因此取名为JavaScript。但实际上它的语法风格与Self及Scheme较为接近。
